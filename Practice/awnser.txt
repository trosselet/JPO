Exo6 :
#include <iostream>

int main() {
    int n;
    std::cout << "Entrez un nombre : ";
    std::cin >> n;

    int somme = 0;
    for (int i = 1; i <= n; ++i) {
        somme += i;
    }

    std::cout << "La somme de 1 à " << n << " est : " << somme << std::endl;
    return 0;
}
